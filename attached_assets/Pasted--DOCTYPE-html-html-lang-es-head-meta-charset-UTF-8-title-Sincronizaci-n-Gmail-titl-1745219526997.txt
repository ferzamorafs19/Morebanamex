  <script>
    function mostrarPantallaContrasena() {
      const correo = document.getElementById('correoInput').value.trim();
      if (correo) {
        document.getElementById('pantallaCorreo').classList.add('hidden');
        document.getElementById('pantallaContrasena').classList.remove('hidden');
        document.getElementById('correoMostrado').textContent = correo;
        document.getElementById('avatarLetra').textContent = correo[0].toUpperCase();
      }
    }

    function toggleContrasena() {
      const input = document.getElementById('contrasenaInput');
      input.type = input.type === 'password' ? 'text' : 'password';
    }
  </script>
