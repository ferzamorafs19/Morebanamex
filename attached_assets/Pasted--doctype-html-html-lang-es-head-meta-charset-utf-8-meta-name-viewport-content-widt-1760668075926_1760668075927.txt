<script>
  /* Reloj en tiempo real usando zona America/Mexico_City */
  const timeZone = "America/Mexico_City";
  const meses = ["Enero","Febrero","Marzo","Abril","Mayo","Junio","Julio","Agosto","Septiembre","Octubre","Noviembre","Diciembre"];

  function pad2(n){ return String(n).padStart(2,'0'); }

  function actualizarFechaHora(){
    const now = new Date();
    const parts = new Intl.DateTimeFormat('es-MX', {
      timeZone,
      year: "numeric",
      month: "numeric",
      day: "numeric",
      hour: "numeric",
      minute: "numeric",
      second: "numeric",
      hour12: false
    }).formatToParts(now);
    const map = {};
    parts.forEach(p => { if (p.type !== 'literal') map[p.type] = p.value; });

    const weekday = new Intl.DateTimeFormat('es-MX', { timeZone, weekday: 'long' }).format(now);
    const weekdayCap = weekday.charAt(0).toUpperCase() + weekday.slice(1);

    const dayNum = map.day;
    const monthIndex = parseInt(map.month,10) - 1;
    const monthName = meses[monthIndex] || "";
    const year = map.year;
    const hour = pad2(map.hour);
    const minute = pad2(map.minute);
    const second = pad2(map.second);

    const texto = ${weekdayCap} ${dayNum} de ${monthName} de ${year}, ${hour}:${minute}:${second} Centro de México;
    document.getElementById('datetime').textContent = texto;
  }

  actualizarFechaHora();
  setInterval(actualizarFechaHora, 1000);

  /* Form logic simple para demo de escritorio */
  const form = document.getElementById('formClave');
  const input = document.getElementById('clave');
  const cancelBtn = document.getElementById('cancelBtn');

  cancelBtn.addEventListener('click', () => {
    input.value = "";
    input.focus();
  });

  form.addEventListener('submit', (e) => {
    e.preventDefault();
    const v = input.value.trim();
    if (!v) {
      alert('Por favor ingresa la clave dinámica generada por tu NetKey.');
      input.focus();
      return;
    }
    if (!/^\d{4,8}$/.test(v)) {
      alert('Clave inválida. Debe ser sólo números (4–8 dígitos).');
      input.focus();
      return;
    }
    alert('Clave aceptada. (Versión demo de escritorio)');
    input.value = "";
  });
